const TEAMS_URL = './standings/teams.json'; const TICKETS_URL = './data/tickets.json'; function esc(s){ return String(s).replace(/[&<>\"]+/g, m => ({'&':'&','<':'<','>':'>','"':'"'}[m] || m)); } function renderStandings(rows){ const tbody = document.querySelector('#standingsTable tbody'); tbody.innerHTML = ''; for(const r of rows){ const teams = (r.teams||[]).map(t => `${esc(t)}`).join(''); const tr = document.createElement('tr'); tr.innerHTML = ` ${r.position} ${r.ticket} ${esc(r.owner)} ${teams} ${r.goals} ${r.gd} `; tbody.appendChild(tr); } } function renderTeams(rows){ const tbody = document.querySelector('#teamsTable tbody'); tbody.innerHTML = ''; for(const r of rows){ const tr = document.createElement('tr'); tr.innerHTML = ` ${esc(r.team)} ${r.gf} ${r.ga} ${r.gd} ${r.played} `; tbody.appendChild(tr); } } function buildStandings(tickets, teamsData){ const teamMap = new Map(); for(const t of (teamsData.teams || [])) teamMap.set(t.team, t); const rows = (tickets || []).map(tk => { const goals = tk.teams.reduce((s,t)=> s + (teamMap.get(t)?.gf || 0), 0); const gd = tk.teams.reduce((s,t)=> s + (teamMap.get(t)?.gd || 0), 0); return { ticket:tk.ticket, owner:tk.owner, teams:tk.teams, goals, gd, rank:(tk.rank ?? tk.ticket) }; }); rows.sort((a,b)=>{ if(b.goals!==a.goals) return b.goals-a.goals; if(b.gd!==a.gd) return b.gd-a.gd; return b.rank-a.rank; }); rows.forEach((r,i)=> r.position=i+1); return rows; } async function load(){ const [teamsData, tickets] = await Promise.all([ fetch(TEAMS_URL,{cache:'no-store'}).then(r=>r.json()), fetch(TICKETS_URL,{cache:'no-store'}).then(r=>r.json()) ]); document.getElementById('updated').textContent = teamsData.generated_at ? `Last updated: ${teamsData.generated_at}` : 'Last updated: unknown'; renderTeams(teamsData.teams || []); renderStandings(buildStandings(tickets, teamsData)); } load(); setInterval(load,120000);
